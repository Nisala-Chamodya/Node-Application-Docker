1.create docker image
docker build -t node-app-img .

2.show images
docker images

3.create docker container
docker run --name new-node-app-con -p 4000:5000 node-app-img


4. create docker image with tag (versioning)
docker build -t node-app-img:v1 .

5. create docker container volune with nodemon
docker run --name new-node-app-con -p 4000:5000 --rm -v ${PWD}:/app -w /app -e CHOKIDAR_USEPOLLING=true node-app-img:v1

6. push images to docker hub

docker tag node-app-img:v1 nisalachamodya/node-app-img:v1
docker images
docker push nisalachamodya/node-app-img:v1

7. clear all docker images and containers
docker system prune -a
